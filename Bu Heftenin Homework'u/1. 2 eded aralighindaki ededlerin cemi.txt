using System;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Xahish olunur Bashlanghic ededi daxil edesiniz: ");
            int StartPoint = int.Parse(Console.ReadLine());

            Console.WriteLine("Xahish olunur Bitish ededi daxil edesiniz: ");
            int EndPoint = int.Parse(Console.ReadLine());

            int Sum = 0;
            for (int i = StartPoint; i <= EndPoint; i++)
            {
                    Sum = Sum + i;
            }
            Console.WriteLine($"Iki eded aralighindaki ededlerin Cemi {Sum}");
        }
    }
}
